'use strict';
          document.addEventListener("DOMContentLoaded", () => {
              const movieDB = {
                  movies: [
                      "Логан",
                      "Лига справедливости",
                      "Ла-ла лэнд",
                      "Одержимость",
                      "Скотт Пилигрим против..."
                  ]
              };
              const advertiseBlock = document.querySelectorAll(".promo__adv img"),
              promoGenre = document.querySelector(".promo__genre"),
              promoBG = document.querySelector(".promo__bg"),
              filmList = document.querySelector(".promo__interactive-list"),
              form = document.querySelector("form.add"),
              input = form.querySelector(".adding__input"),
              checkbox = form.querySelector("[type = 'checkbox']");
              form.addEventListener("submit", (e) => {
                  e.preventDefault();
                  let film = input.value;
                  const favoriteFilm = checkbox.checked;
                  if (film) {
                      if (film.length > 21) {
                          film = `${film.slice(0, 21)}...`;
                      }
                      if (favoriteFilm) {
                          console.log("Добавляем любимый фильм");
                      }
                      movieDB.movies.push(film);
                      sortArr(movieDB.movies);
                      createMovieList(movieDB.movies, filmList);
                      console.log(movieDB.movies);
                      e.target.reset();
                  }
              });
              const deleteAdv = (arr) => {
                  arr.forEach((item) => {
                      item.remove();
                  });
              };
              const makeChanges = () => {
                  promoGenre.textContent = "драма";
                  promoBG.style.backgroundImage = "url('img/bg.jpg')";
              };
              const sortArr = (arr) => {
                  arr.sort();
              };
              function createMovieList (films, parent) {
                  parent.innerHTML = "";
                  sortArr(films);
                  films.forEach((item, i) => {
                      parent.innerHTML += `<li class="promo__interactive-item">${i + 1} ${item}
                                              <div class="delete"></div>
                                              </li>`;
                  });
                  document.querySelectorAll(".delete").forEach((btn, i) => {
                      btn.addEventListener("click", () => {
                          btn.parentElement.remove();
                          movieDB.movies.splice(i, 1);

                          createMovieList(films, parent);
                      });
                  });
              }
              deleteAdv(advertiseBlock);
              makeChanges();
              createMovieList(movieDB.movies, filmList);
          });
